import { collection, doc, setDoc , getDocs, getDoc} from 'firebase/firestore';
import { db, storage } from '../../firebase';

import {
  FormControl,

const Test = () => {
  const [batches, setBatches] = useState([]);
  const [students, setStudents] = useState([]);
  const [selectedBatch, setSelectedBatch] = useState('');
  const [selectedStudents, setSelectedStudents] = useState([]);

  // const [selectedStudents, setSelectedStudents] = useState([]);

// Fetch selected students from Firestore when the component mounts
useEffect(() => {
  const fetchSelectedStudents = async () => {
    const trainerName = 'John Doe'; // replace with trainer name
    const trainerRef = doc(db, 'trainers', trainerName);
    const trainerSnapshot = await getDoc(trainerRef);
    if (trainerSnapshot.exists()) {
      const { students } = trainerSnapshot.data();
      const selectedStudents = students.map((student) => student.studentId);
      setSelectedStudents(selectedStudents);
    }
  };

  fetchSelectedStudents();
}, []);



useEffect(() => {
    const fetchBatches = async () => {
      const batchesRef = collection(db, 'batches');
      const batchesSnapshot = await getDocs(batchesRef);
      const batchList = batchesSnapshot.docs.map((doc) => ({
        id: doc.id,
        name: doc.data().name,
      }));
      setBatches(batchList);
    };

    fetchBatches();
  }, []);

  const handleBatchChange = async (event) => {
    const batchId = event.target.value;
    setSelectedBatch(batchId);

    const studentsRef = collection(db, 'batches', batchId, 'students');
    const studentsSnapshot = await getDocs(studentsRef);
    const studentList = studentsSnapshot.docs.map((doc) => ({
      id: doc.id,
      name: doc.data().name,
    }));
    setStudents(studentList);
  };



  const handleStudentChange = (event, studentId) => {
    const checked = event.target.checked;
  
    if (checked) {
      setSelectedStudents((prevSelectedStudents) => [...prevSelectedStudents, studentId]);
    } else {
      setSelectedStudents((prevSelectedStudents) =>
        prevSelectedStudents.filter((id) => id !== studentId)
      );
    }
  };
  
  

const handleAddClick = async () => {
  const trainerName = 'John Doe'; // replace with trainer name
  const trainerRef = doc(db, 'trainers', trainerName);

  const selectedStudentsData = selectedStudents.map((studentId) => ({
    studentId,
    batchId: selectedBatch, // replace with the actual batch ID
  }));

  try {
    await setDoc(trainerRef, { students: selectedStudentsData }, { merge: true });
    console.log('Students added successfully!');
  } catch (error) {
    console.error('Error adding students:', error);
  }
};



  return (
    <div>
      <FormControl sx={{ m: 1, minWidth: 120 }}>
        <InputLabel id="batch-label">Batch</InputLabel>
        <Select
          labelId="batch-label"
          id="batch"
          value={selectedBatch}
          onChange={handleBatchChange}
        >
          {batches.map((batch) => (
            <MenuItem key={batch.id} value={batch.id}>
              {batch.name}
            </MenuItem>
          ))}
        </Select>
      </FormControl>
      <FormControl sx={{ m: 1, minWidth: 120 }}>
        <InputLabel id="student-label" shrink>Students</InputLabel>
        <Select
          labelId="student-label"
          id="student"
          multiple
          value={selectedStudents}
          onChange={handleStudentChange}
          renderValue={(selected) => selected.join(', ')}
        >
          {students.map((student) => (

                    <FormControlLabel
                      key={student.id}
                      control={
                      <Checkbox
                        checked={selectedStudents.includes(student.id)}
                        onChange={(event) => handleStudentChange(event, student.id)}
                      />
                      }
                      label={student.name}
                    />
          ))}
        </Select>
      </FormControl>
      <Button variant="contained" onClick={handleAddClick}>Add</Button>
    </div>
  );
};

export default Test;
